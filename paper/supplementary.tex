\section{Dataset Metadata}

\subsection{Eurostat High Density Clusters (HDENS-CLST) 2021}

\begin{itemize}
  \item \textbf{Source}: European Commission Joint Research Centre (JRC)
  \item \textbf{URL}: \url{https://ghsl.jrc.ec.europa.eu/}
  \item \textbf{Resolution}: 1km × 1km grid
  \item \textbf{Definition}: Contiguous cells with $\geq$1,500 inhabitants/km$^2$ and cumulative population $\geq$50,000
  \item \textbf{Reference Year}: 2021
  \item \textbf{Coverage}: Pan-European
  \item \textbf{License}: CC BY 4.0
  \item \textbf{Processing}: Vectorised, filtered to continental Europe (EPSG:3035), UK excluded
\end{itemize}

\subsection{Overture Maps Foundation -- 2024 Release}

\begin{itemize}
  \item \textbf{Source}: Overture Maps Foundation
  \item \textbf{URL}: \url{https://overturemaps.org/}
  \item \textbf{Release Date}: 2024-07-22 (alpha release)
  \item \textbf{Themes Used}:
    \begin{itemize}
      \item Places (POIs): 2,000+ categories
      \item Buildings: Footprints with attributes
      \item Transportation: Road network (connectors and segments)
    \end{itemize}
  \item \textbf{Access Method}: STAC (SpatioTemporal Asset Catalog)
  \item \textbf{License}: CDLA-Permissive-2.0 / ODbL (depending on source)
  \item \textbf{Processing}: Clipped to urban boundaries, transformed to EPSG:3035
\end{itemize}

\subsection{Copernicus Urban Atlas 2018}

\begin{itemize}
  \item \textbf{Source}: European Environment Agency / Copernicus Land Monitoring Service
  \item \textbf{URL}: \url{https://land.copernicus.eu/local/urban-atlas}
  \item \textbf{Reference Year}: 2018
  \item \textbf{Resolution}: Minimum Mapping Unit (MMU) 0.25 ha for artificial surfaces
  \item \textbf{Classification}: 27 land cover/use classes
  \item \textbf{Coverage}: 788 Functional Urban Areas (FUAs) across EU
  \item \textbf{License}: Copernicus data policy (free and open access)
  \item \textbf{Processing}: Extracted urban blocks, green spaces; clipped to HDENS boundaries
\end{itemize}

\subsection{Copernicus Street Tree Layer 2018}

\begin{itemize}
  \item \textbf{Source}: European Environment Agency / Copernicus Land Monitoring Service
  \item \textbf{URL}: \url{https://land.copernicus.eu/local/urban-atlas/street-tree-layer-stl-2018}
  \item \textbf{Reference Year}: 2018
  \item \textbf{Resolution}: 0.5m spatial resolution
  \item \textbf{Method}: Very High Resolution imagery interpretation
  \item \textbf{Coverage}: Street trees in Urban Atlas FUAs
  \item \textbf{License}: Copernicus data policy
  \item \textbf{Processing}: Clipped to HDENS boundaries; used for tree proximity calculations
\end{itemize}

\subsection{Copernicus Building Height 2012}

\begin{itemize}
  \item \textbf{Source}: European Environment Agency / Copernicus Land Monitoring Service
  \item \textbf{URL}: \url{https://land.copernicus.eu/local/urban-atlas/building-height-2012}
  \item \textbf{Reference Year}: 2012
  \item \textbf{Resolution}: 10m × 10m raster
  \item \textbf{Method}: Digital Height Model (DHM) derived from stereo imagery
  \item \textbf{Units}: Metres above ground
  \item \textbf{Coverage}: Urban Atlas FUAs
  \item \textbf{License}: Copernicus data policy
  \item \textbf{Processing}: Sampled at building footprint centroids
\end{itemize}

\subsection{Eurostat Census Grid 2021}

\begin{itemize}
  \item \textbf{Source}: Eurostat GEOSTAT project
  \item \textbf{URL}: \url{https://ec.europa.eu/eurostat/web/gisco/geodata/population-distribution/geostat}
  \item \textbf{Reference Year}: 2021 (Census round)
  \item \textbf{Resolution}: 1km × 1km grid (INSPIRE compliant)
  \item \textbf{Variables}: Population, employment, demographics, migration
  \item \textbf{Coverage}: EU27 + EFTA
  \item \textbf{License}: CC BY 4.0
  \item \textbf{Processing}: Interpolated to network nodes via grid cell centroids
\end{itemize}

\section{Processing Parameters}

\subsection{Network Processing}

\begin{itemize}
  \item \textbf{CRS}: EPSG:3035 (ETRS89-LAEA Europe)
  \item \textbf{Decomposition}: 80m segments ($\approx$1-minute walk at 80m/min)
  \item \textbf{Cleaning tolerance}: 8m (consolidation of degree-2 nodes)
  \item \textbf{Component threshold}: 100 nodes minimum
  \item \textbf{Level-aware processing}: Enabled (prevents incorrect bridge merging)
  \item \textbf{Edge classification}: Based on Overture connector/segment attributes
\end{itemize}

\subsection{Centrality Computation}

\begin{itemize}
  \item \textbf{Method}: Segment-based (dual graph)
  \item \textbf{Thresholds}: [400, 800, 1200, 1600, 4800, 9600] metres
  \item \textbf{Walking times}: [5, 10, 15, 20, 60, 120] minutes
  \item \textbf{Walking speed}: 80 m/min ($\approx$4.8 km/h)
  \item \textbf{Beta decay}: Exponential ($e^{-\beta \cdot d}$)
  \item \textbf{Min threshold weight}: 0.01832 (default cityseer value)
\end{itemize}

\subsection{Accessibility Computation}

\begin{itemize}
  \item \textbf{Thresholds}: [400, 800, 1200, 1600] metres
  \item \textbf{Walking times}: [5, 10, 15, 20] minutes
  \item \textbf{Max assignment distance}: 100m (POI to nearest network edge)
  \item \textbf{Metrics}: Unweighted count, distance-weighted count, nearest distance
  \item \textbf{Weight function}: Exponential decay
  \item \textbf{Hill diversity}: q = [0, 1, 2] for richness, Shannon, Simpson
\end{itemize}

\subsection{POI Saturation Assessment}

\begin{itemize}
  \item \textbf{Grid resolution}: 1km × 1km (Eurostat Census Grid)
  \item \textbf{Population scales}:
    \begin{itemize}
      \item Local: 1km radius
      \item Intermediate: 5km radius
      \item Large: 10km radius
    \end{itemize}
  \item \textbf{Model}: Random Forest Regressor (sklearn)
    \begin{itemize}
      \item Trees: 100
      \item Max depth: 20
      \item Min samples split: 5
      \item Random state: 42
    \end{itemize}
  \item \textbf{Transformation}: Log-space ($\log(POI+1)$ and $\log(pop+1)$)
  \item \textbf{Z-score}: Standardised residuals from log-space predictions
  \item \textbf{Quadrant threshold}: Median standard deviation across categories
\end{itemize}

\section{Computational Requirements}

\subsection{Hardware}

\begin{itemize}
  \item \textbf{Minimum RAM}: 16 GB (32 GB recommended for largest cities)
  \item \textbf{Storage}: $\approx$500 GB for all 699 cities (complete pipeline outputs)
  \item \textbf{Processing time}: $\approx$2-5 hours per city (depending on size and complexity)
\end{itemize}

\subsection{Parallelisation}

\begin{itemize}
  \item \textbf{City-level}: Fully independent (embarrassingly parallel)
  \item \textbf{Metric computation}: Single-threaded cityseer Rust backend
  \item \textbf{Recommended strategy}: Process multiple cities simultaneously
\end{itemize}

\section{Data Quality Notes}

\subsection{Known Limitations}

\begin{itemize}
  \item \textbf{Temporal mismatch}: Source datasets span 2012-2024
    \begin{itemize}
      \item Building heights: 2012
      \item Urban Atlas: 2018
      \item Census: 2021
      \item Overture: 2024
    \end{itemize}
  \item \textbf{POI completeness}: Variable across regions (see saturation assessment)
  \item \textbf{Building heights}: Not available for all FUAs; missing values set to estimated median
  \item \textbf{Network topology}: Minor inconsistencies at administrative boundaries
\end{itemize}

\subsection{Quality Assurance}

\begin{itemize}
  \item \textbf{Network validation}: Automatic cleaning via cityseer (8m tolerance)
  \item \textbf{Topology checks}: Connected component analysis (min 100 nodes)
  \item \textbf{Attribute validation}: Range checks on computed metrics
  \item \textbf{Visual inspection}: Sample cities checked for geometric accuracy
  \item \textbf{Saturation assessment}: POI quality validated via multi-scale regression
\end{itemize}
